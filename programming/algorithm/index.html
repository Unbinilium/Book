# Point to Point

Rotate P(x, y) rounded the O(o_x, o_y) with angle theta, the new P' is (x_r, y_r).

def rotate(x, y, o_x, o_y, theta):
    x_r = math.cos(theta) * (x - o_x) - math.sin(theta) * (y - o_y) + o_x
    y_r = math.sin(theta) * (x - o_x) + math.cos(theta) * (y - o_y) + o_y
    return [x_r, y_r]
